92  glTexCoord2f(1.0*(pos.pos_x-0.5+wo+w)/C, 1.0*(pos.pos_y-0.5+pos.rows)/C); glVertex3f(0.5+(sx+i.x)/64.0+pos.left+w-1, 0.5+(sy+i.y+32)/64-pos.top+pos.rows-1, 0);

133  void showLayout(const TextLayout_c & l, int sx, int sy, SubPixelArrangement sp, imageDrawerOpenGL_c * images)
134  {
135  // make sure that all required glyphs are on our glyph texture, when necessary recreate it freshly
136  // bind the texture
137  for (auto & i : l.getData())
138  {
139  switch (i.command)
140  {
141  case CommandData_c::CMD_GLYPH:
142  // when subpixel placement is on we always create all 3 required images
143  cache.getGlyph(i.font, i.glyphIndex, sp, i.blurr);
144  break;
145  case CommandData_c::CMD_RECT:
146  if (i.blurr > 0)
147  cache.getRect(i.w, i.h, sp, i.blurr);
148  break;
149 
150  default:
151  break;
152  }
153  }
154 
155  glBindTexture( GL_TEXTURE_2D, glTextureId );
156 
157  if (cache.getVersion() != uploadVersion)
158  {
159  uploadVersion = cache.getVersion();
160  glTexImage2D( GL_TEXTURE_2D, 0, GL_ALPHA, C, C, 0, GL_ALPHA, GL_UNSIGNED_BYTE, cache.getData() );
161  }
162 
163  glEnable(GL_BLEND);
164  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
165 
166  // output the layout
167  for (auto & i : l.getData())
168  {
169  switch (i.command)
170  {
171  case CommandData_c::CMD_GLYPH:
172  {
173  glEnable(GL_TEXTURE_2D);
174 
175  if (i.blurr > cache.blurrmax)
176  {
177  drawGlyph(i, sx, sy, sp, 0);
178  }
179  else
180  {
181  switch (sp)
182  {
183  case SUBP_RGB:
184  glColorMask(GL_TRUE, GL_FALSE, GL_FALSE, GL_FALSE); drawGlyph(i, sx, sy, sp, 1);
185  glColorMask(GL_FALSE, GL_TRUE, GL_FALSE, GL_FALSE); drawGlyph(i, sx, sy, sp, 2);
186  glColorMask(GL_FALSE, GL_FALSE, GL_TRUE, GL_FALSE); drawGlyph(i, sx, sy, sp, 3);
187  glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
188  break;
189 
190  case SUBP_BGR:
191  glColorMask(GL_FALSE, GL_FALSE, GL_TRUE, GL_FALSE); drawGlyph(i, sx, sy, sp, 1);
192  glColorMask(GL_FALSE, GL_TRUE, GL_FALSE, GL_FALSE); drawGlyph(i, sx, sy, sp, 2);
193  glColorMask(GL_TRUE, GL_FALSE, GL_FALSE, GL_FALSE); drawGlyph(i, sx, sy, sp, 3);
194  glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
195  break;
196 
197  default:
198  drawGlyph(i, sx, sy, sp, 0);
199  break;
200  }
201  }
202  }
203  break;
204 
205  case CommandData_c::CMD_RECT:
206  {
207  if (i.blurr == 0)
208  {
209  glDisable(GL_TEXTURE_2D);
210  Color_c c = gamma.forward(i.c);
211  glBegin(GL_QUADS);
212  glColor3f(c.r()/255.0, c.g()/255.0, c.b()/255.0);
213  glVertex3f((sx+i.x +32)/64+0.5, (sy+i.y +32)/64+0.5, 0);
214  glVertex3f((sx+i.x+i.w+32)/64+0.5, (sy+i.y +32)/64+0.5, 0);
215  glVertex3f((sx+i.x+i.w+32)/64+0.5, (sy+i.y+i.h+32)/64+0.5, 0);
216  glVertex3f((sx+i.x +32)/64+0.5, (sy+i.y+i.h+32)/64+0.5, 0);
217  glEnd();
218  }
219  else
220  {
221  glEnable(GL_TEXTURE_2D);
222  auto pos = cache.getRect(i.w, i.h, sp, i.blurr);
223  glBegin(GL_QUADS);
224  Color_c c = gamma.forward(i.c);
225  glColor3f(c.r()/255.0, c.g()/255.0, c.b()/255.0);
226  glTexCoord2f(1.0*(pos.pos_x-0.5)/C, 1.0*(pos.pos_y-0.5)/C); glVertex3f(0.5+(sx+i.x)/64.0+pos.left-1, 0.5+(sy+i.y+32)/64-pos.top-1, 0);
227  glTexCoord2f(1.0*(pos.pos_x-0.5+pos.width)/C, 1.0*(pos.pos_y-0.5)/C); glVertex3f(0.5+(sx+i.x)/64.0+pos.left+pos.width-1, 0.5+(sy+i.y+32)/64-pos.top-1, 0);
228  glTexCoord2f(1.0*(pos.pos_x-0.5+pos.width)/C, 1.0*(pos.pos_y-0.5+pos.rows)/C); glVertex3f(0.5+(sx+i.x)/64.0+pos.left+pos.width-1, 0.5+(sy+i.y+32)/64-pos.top+pos.rows-1, 0);
229  glTexCoord2f(1.0*(pos.pos_x-0.5)/C, 1.0*(pos.pos_y-0.5+pos.rows)/C); glVertex3f(0.5+(sx+i.x)/64.0+pos.left-1, 0.5+(sy+i.y+32)/64-pos.top+pos.rows-1, 0);
230  glEnd();
231  }
232  }

234

238  break;
239  }
240  }
